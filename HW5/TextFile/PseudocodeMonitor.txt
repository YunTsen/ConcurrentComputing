    MonitorBegin();    
    queuingCannbals.push_back(c);
    cannbals++;//多了一隻食人怪在排隊
    if(cannbals+missionaries>=3){//有三個以上乘客在等
        canPick.signal;
    }
    MonitorEnd();
    return;
}

    return;
}
